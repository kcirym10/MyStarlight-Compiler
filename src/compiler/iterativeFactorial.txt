program factorial;

func int iterativeFactorial(int n)
var
    int i, acum;
{
    acum = 1;
    i = 1;
    while (i <= n){
        acum = acum * i;
        i = i + 1;
    }
    return(acum);
}

main()
var
    int numberUser;
{
    print("Enter a number to calculate the factorial");
    read(numberUser);

    print("Iterative version's result", iterativeFactorial(numberUser));
}
